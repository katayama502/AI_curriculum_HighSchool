バリデーションとエラーハンドリング

バリデーションの2層構造

フロントエンドバリデーション（UX向上）
入力中にリアルタイムでエラー表示。ユーザーに早く気づかせる
バックエンドバリデーション（必須・セキュリティ）
フロントを迂回した攻撃にも対応。DBに不正データが入るのを防ぐ
// フロントエンドのバリデーション例
const [errors, setErrors] = useState({});

const validate = (data) => {
const errs = {};
if (!data.title.trim())
errs.title = 'タイトルを入力してください';
if (data.title.length > 100)
errs.title = '100文字以内で入力してください';
return errs;
};

// JSX で表示
{errors.title && <p style={{color:'red'}}>{errors.title}</p>}
フロントのバリデーションだけでは不十分！バックエンドでも必ず確認する

// バックエンドのバリデーション
router.post('/', authMiddleware, (req, res) => {
const { title, author } = req.body;

// バリデーション
if (!title?.trim())
return res.status(400)
.json({ error: 'タイトルは必須です' });
if (title.length > 100)
return res.status(400)
.json({ error: 'タイトルは100文字以内' });

// DB保存
try {
const result = db.prepare('...').run(...);
res.status(201).json({ id: result.lastInsertRowid });
} catch (err) {
console.error(err);
res.status(500).json({ error: '保存に失敗しました' });
}
});

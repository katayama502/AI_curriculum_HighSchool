JavaScriptの基本文法

// 変数の宣言
let name = "太郎";
const age = 16;

// 条件分岐
if (age >= 18) {
  console.log("大人");
} else {
  console.log("未成年");
}

// ループ
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// 関数
function greet(name) {
  return `こんにちは、${name}！`;
}
let/const
変数宣言
let = 変更可能、const = 変更不可（定数）
if/else
条件分岐
条件が true/false で処理を分ける
for
繰り返し処理
同じ処理を指定回数繰り返す
function
関数定義
処理をまとめて再利用可能にする
console.log
デバッグ出力
ブラウザの開発者ツールに値を表示

💻 Expressの基本コード
const express = require('express');
const app = express();
app.use(express.json());

// GETエンドポイント
app.get('/api/users', (req, res) => {
  res.json([{ id: 1, name: '太郎' }]);
});

// POSTエンドポイント
app.post('/api/users', (req, res) => {
  const { name } = req.body;
  res.json({ message: `${name}を追加` });
});

app.listen(3000);

💻 Reactの基本コード
import React, { useState } from 'react';

// コンポーネント（部品）を定義
function Counter() {
  // 状態を定義（count = 初期値0）
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>カウント: {count}</p>
      <button
        onClick={() => setCount(count + 1)}
      >+1</button>
    </div>
  );
}

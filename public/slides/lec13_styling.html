💻 Tailwind CSSの例
// Tailwind CSSでスタイリング
function Card({ title, text }) {
  return (
    <div className="bg-white p-6 shadow-md">
      <h2 className="text-2xl font-bold text-blue-600">
        {title}
      </h2>
      <p className="text-gray-600 mt-2">
        {text}
      </p>
    </div>
  );
}

フロントエンドとバックエンドの接続

🏗️ フルスタックアーキテクチャ
⚛️ React フロントエンド
↕ HTTP / JSON
🔌 Express.js API サーバー
↕ SQL
🗄️ SQLite データベース
💻 ReactからAPIを呼び出す
import { useState, useEffect } from 'react';

function UserList() {
  const [users, setUsers] = useState([]);

  useEffect(() => {
    // バックエンドAPIを呼び出す
    fetch('/api/users')
      .then(res => res.json())
      .then(data => setUsers(data));
  }, []);

  return (
    <ul>
      {users.map(u => <li>{u.name}</li>)}
    </ul>
  );
}

CRUD APIの実装パターン

GET /api/books — 一覧取得
router.get('/', authMiddleware, (req, res) => {
const books = db.prepare(
'SELECT * FROM books
WHERE user_id=? ORDER BY id DESC'
).all(req.user.userId);
res.json(books);
});
POST /api/books — 追加
router.post('/', authMiddleware, (req, res) => {
const { title, author } = req.body;
if (!title) return res.status(400)
.json({ error: 'タイトルは必須' });
const result = db.prepare(
'INSERT INTO books (title,author,user_id)
VALUES (?,?,?)'
).run(title, author, req.user.userId);
res.status(201).json({ id: result.lastInsertRowid });
});
PUT /api/books/:id — 更新
router.put('/:id', authMiddleware, (req, res) => {
const { title, author } = req.body;
db.prepare(
'UPDATE books SET title=?, author=?
WHERE id=? AND user_id=?'
).run(title, author, req.params.id,
req.user.userId);
res.json({ message: '更新しました' });
});
DELETE /api/books/:id — 削除
router.delete('/:id', authMiddleware, (req, res) => {
db.prepare(
'DELETE FROM books
WHERE id=? AND user_id=?'
).run(req.params.id, req.user.userId);
res.json({ message: '削除しました' });
});
ポイント： WHERE user_id=? を付けることで、他人のデータを操作できないようにする（認可）

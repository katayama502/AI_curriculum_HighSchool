// ボタンをクリックしたとき const btn = document.getElementById("myBtn"); btn.addEventListener("click", function() { alert("ボタンが押されました！"); }); // 入力フォームの値を取得 const input = document.getElementById("nameInput"); input.addEventListener("input", function(e) { const val = e.target.value; document.getElementById("display") .textContent = "こんにちは、" + val; });

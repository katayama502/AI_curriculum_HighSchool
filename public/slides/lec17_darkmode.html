CSS変数で切り替える

/* ライトモード（デフォルト）*/
:root {
--bg: #F8F9FA;
--card-bg: #ffffff;
--text: #212529;
--sub-text: #868e96;
}

/* ダークモード */
@media (prefers-color-scheme: dark) {
:root {
--bg: #25262b;
--card-bg: #2c2e33;
--text: #c1c2c5;
--sub-text: #5c5f66;
}
}

/* 使い方 */
body { background: var(--bg); }
.card { background: var(--card-bg); }

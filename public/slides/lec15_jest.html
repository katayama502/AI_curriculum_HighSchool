Jest：JavaScriptの単体テスト

テストの書き方

// utils.js — テスト対象の関数
function add(a, b) { return a + b; }
function isEmpty(str) { return str.trim() === ''; }
module.exports = { add, isEmpty };
// utils.test.js — テストファイル
const { add, isEmpty } = require('./utils');

describe('add関数', () => {
  test('2+3は5になる', () => {
    expect(add(2, 3)).toBe(5);
  });
  test('負の数も足せる', () => {
    expect(add(-1, 1)).toBe(0);
  });
});

test('空文字はtrueを返す', () => {
  expect(isEmpty('')).toBe(true);
  expect(isEmpty('hello')).toBe(false);
});
✓ 2+3は5になる
✓ 負の数も足せる
✓ 空文字はtrueを返す
Tests: 3 passed, 3 total

よく使うマッチャー

値の確認
.toBe(value) 厳密一致
.toEqual(obj) オブジェクト
.toBeTruthy() 真偽値
.toBeNull() null確認
数値・文字列
.toBeGreaterThan(n)
.toContain(str)
.toHaveLength(n)
エラー確認
.toThrow() エラーが出るか
.not.toBe() 否定形

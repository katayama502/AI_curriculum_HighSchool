状態管理とuseState

🔄 useStateの使い方
import { useState } from 'react';

function TodoApp() {
  const [todos, setTodos] = useState([]);
  const [input, setInput] = useState("");

  const addTodo = () => {
    setTodos([...todos, input]);
    setInput("");
  };

  return (
    <div>
      <input value={input}
        onChange={e => setInput(e.target.value)} />
      <button onClick={addTodo}>追加</button>
    </div>
  );
}
